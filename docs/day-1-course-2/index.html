1. 포인트 클라우드 기초 (참고)

이어지는 실습들을 위해 깊이 이미지(Depth Image)로부터 포인트 클라우드를 생성하는 기초 원리를 다룹니다. 카메라 내부 파라미터(Intrinsic)를 이용해 2D 픽셀 좌표와 깊이 값을 3D 공간 좌표로 변환합니다. 본 차시에서는 아래의 함수가 미리 정의되어 있다고 가정하고 실습을 진행합니다.

📖 이론 설명

# 깊이 이미지에서 포인트 클라우드를 생성하는 기본 함수 (참고용)
def create_point_cloud_from_depth(depth_img, camera_matrix):
    h, w = depth_img.shape
    fx, fy = camera_matrix[0, 0], camera_matrix[1, 1]
    cx, cy = camera_matrix[0, 2], camera_matrix[1, 2]
    
    # 픽셀 좌표 그리드 생성
    u, v = np.meshgrid(np.arange(w), np.arange(h))
    
    # Z값(깊이)이 0보다 큰 유효한 픽셀만 추출
    valid = depth_img > 0
    z = depth_img[valid]
    
    # 핀홀 카메라 모델 역투영 수식: X = (u - cx) * Z / fx
    x = (u[valid] - cx) * z / fx
    y = (v[valid] - cy) * z / fy
    
    # (N, 3) 형태의 포인트 클라우드 배열 반환
    points = np.vstack((x, y, z)).T
    return points

def get_line_intersection(p1, p2, p3, p4):
    """
    두 선분 (p1-p2), (p3-p4)의 무한 연장선 교차점(소실점)을 반환합니다.
    """
    x1, y1 = p1; x2, y2 = p2
    x3, y3 = p3; x4, y4 = p4

    denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)
    if denom == 0: 
        return None # 평행할 경우
    
    px = ((x1*y2 - y1*x2)*(x3 - x4) - (x1 - x2)*(x3*y4 - y3*x4)) / denom
    py = ((x1*y2 - y1*x2)*(y3 - y4) - (y1 - y2)*(x3*y4 - y3*x4)) / denom
    return np.array([px, py], dtype=np.float32)
